SwiftUI-TextField

July 31, 2019
SwiftUI
Page content

TextField用来添加普通的输入框，一般常用作文本输入。


import SwiftUI
import PlaygroundSupport

struct TextFieldPage: View {
    
    @State var name = ""
    @State var pass = ""
    
    var body: some View {
        VStack {
            TextField(self.$name, placeholder: Text("Name"), onEditingChanged: { (changed) in
                print(changed)
            }) {
                print(self.name)
            }
            .textFieldStyle(.roundedBorder)
                .frame(width: 200)
            SecureField(self.$pass, placeholder: Text("Pass")) {
                print(self.pass)
            }.textFieldStyle(.roundedBorder)
                .frame(width: 200)
            
        }
    }
}

PlaygroundPage.current.liveView = UIHostingController(rootView: TextFieldPage())
