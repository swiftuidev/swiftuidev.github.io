SwiftUI-Text

July 31, 2019
SwiftUI
Page content

Text用来展示一行或多行的文本内容，效果等同于UILabel，但更加优秀. 如果要创建Text， 只需通过Text("Text")即可创建. 也可以为文本添加多项属性，如字体、颜色、阴影、上左下右的间距等。

import SwiftUI
import PlaygroundSupport

struct TextPage: View {
    var body: some View {
        VStack {
            Text("普通的Text")
            Text("有文字颜色的Text").foregroundColor(.blue)
            Text("加粗的Text").bold()
            Text("斜体的Text").italic()
            Text("阴影的Text").shadow(color: .red, radius: 10, x: 0, y: 2)
            Text("Text").fontWeight(.medium)
            Text("Text").font(.system(.largeTitle))
            Text("https://swiftuidev.com").underline().tapAction {
                print("https://swiftuidev.com")
            }
            Text("https://swiftuidev.com").strikethrough()
            Text("Text 用来展示一行或多行的文本内容，效果等同于 UILabel，但更加优秀。 如果要创建 Text， 只需通过 Text(\"Text\") 即可创建； 采用链式语法，也可以为文本添加多项属性，如字体、颜色、阴影、上左下右的间距等。").lineLimit(nil)
        }
    }
}

PlaygroundPage.current.liveView = UIHostingController(rootView: TextPage())
